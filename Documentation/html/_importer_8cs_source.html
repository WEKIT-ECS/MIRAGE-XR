94  private static string GetGLBJson(Stream stream, out long binChunkStart) {
95  byte[] buffer = new byte[12];
96  stream.Read(buffer, 0, 12);
97  // 12 byte header
98  // 0-4 - magic = "glTF"
99  // 4-8 - version = 2
100  // 8-12 - length = total length of glb, including Header and all Chunks, in bytes.
101  string magic = Encoding.Default.GetString(buffer, 0, 4);
102  if (magic != "glTF") {
103  Debug.LogWarning("Input does not look like a .glb file");
104  binChunkStart = 0;
105  return null;
106  }
107  uint version = System.BitConverter.ToUInt32(buffer, 4);
108  if (version != 2) {
109  Debug.LogWarning("Importer does not support gltf version " + version);
110  binChunkStart = 0;
111  return null;
112  }
113  // What do we even need the length for.
114  //uint length = System.BitConverter.ToUInt32(bytes, 8);
115 
116  // Chunk 0 (json)
117  // 0-4 - chunkLength = total length of the chunkData
118  // 4-8 - chunkType = "JSON"
119  // 8-[chunkLength+8] - chunkData = json data.
120  stream.Read(buffer, 0, 8);
121  uint chunkLength = System.BitConverter.ToUInt32(buffer, 0);
122  TextReader reader = new StreamReader(stream);
123  char[] jsonChars = new char[chunkLength];
124  reader.Read(jsonChars, 0, (int) chunkLength);
125  string json = new string(jsonChars);
126 
127  // Chunk
128  binChunkStart = chunkLength + 20;
129  stream.Close();
130 
131  // Return json
132  return json;
133  }

185  private static GameObject LoadInternal(this GLTFObject gltfObject, string filepath, byte[] bytefile, long binChunkStart, ImportSettings importSettings, out AnimationClip[] animations) {
186  CheckExtensions(gltfObject);
187 
188  // directory root is sometimes used for loading buffers from containing file, or local images
189  string directoryRoot = filepath != null ? Directory.GetParent(filepath).ToString() + "/" : null;
190 
191  importSettings.shaderOverrides.CacheDefaultShaders();
192 
193  // Import tasks synchronously
194  GLTFBuffer.ImportTask bufferTask = new GLTFBuffer.ImportTask(gltfObject.buffers, filepath, bytefile, binChunkStart);
195  bufferTask.RunSynchronously();
196  GLTFBufferView.ImportTask bufferViewTask = new GLTFBufferView.ImportTask(gltfObject.bufferViews, bufferTask);
197  bufferViewTask.RunSynchronously();
198  GLTFAccessor.ImportTask accessorTask = new GLTFAccessor.ImportTask(gltfObject.accessors, bufferViewTask);
199  accessorTask.RunSynchronously();
200  GLTFImage.ImportTask imageTask = new GLTFImage.ImportTask(gltfObject.images, directoryRoot, bufferViewTask);
201  imageTask.RunSynchronously();
202  GLTFTexture.ImportTask textureTask = new GLTFTexture.ImportTask(gltfObject.textures, imageTask);
203  textureTask.RunSynchronously();
204  GLTFMaterial.ImportTask materialTask = new GLTFMaterial.ImportTask(gltfObject.materials, textureTask, importSettings);
205  materialTask.RunSynchronously();
206  GLTFMesh.ImportTask meshTask = new GLTFMesh.ImportTask(gltfObject.meshes, accessorTask, bufferViewTask, materialTask, importSettings);
207  meshTask.RunSynchronously();
208  GLTFSkin.ImportTask skinTask = new GLTFSkin.ImportTask(gltfObject.skins, accessorTask);
209  skinTask.RunSynchronously();
210  GLTFNode.ImportTask nodeTask = new GLTFNode.ImportTask(gltfObject.nodes, meshTask, skinTask, gltfObject.cameras);
211  nodeTask.RunSynchronously();
212  GLTFAnimation.ImportResult[] animationResult = gltfObject.animations.Import(accessorTask.Result, nodeTask.Result, importSettings);
213  if (animationResult != null) animations = animationResult.Select(x => x.clip).ToArray();
214  else animations = new AnimationClip[0];
215 
216  foreach (var item in bufferTask.Result) {
217  item.Dispose();
218  }
219 
220  return nodeTask.Result.GetRoot();
221  }

225  private static IEnumerator LoadAsync(string json, string filepath, byte[] bytefile, long binChunkStart, ImportSettings importSettings, Action<GameObject, AnimationClip[]> onFinished, Action<float> onProgress = null) {
226  // Threaded deserialization
227  Task<GLTFObject> deserializeTask = new Task<GLTFObject>(() => JsonConvert.DeserializeObject<GLTFObject>(json));
228  deserializeTask.Start();
229  while (!deserializeTask.IsCompleted) yield return null;
230  GLTFObject gltfObject = deserializeTask.Result;
231  CheckExtensions(gltfObject);
232 
233  // directory root is sometimes used for loading buffers from containing file, or local images
234  string directoryRoot = filepath != null ? Directory.GetParent(filepath).ToString() + "/" : null;
235 
236  importSettings.shaderOverrides.CacheDefaultShaders();
237 
238  // Setup import tasks
239  List<ImportTask> importTasks = new List<ImportTask>();
240 
241  GLTFBuffer.ImportTask bufferTask = new GLTFBuffer.ImportTask(gltfObject.buffers, filepath, bytefile, binChunkStart);
242  importTasks.Add(bufferTask);
243  GLTFBufferView.ImportTask bufferViewTask = new GLTFBufferView.ImportTask(gltfObject.bufferViews, bufferTask);
244  importTasks.Add(bufferViewTask);
245  GLTFAccessor.ImportTask accessorTask = new GLTFAccessor.ImportTask(gltfObject.accessors, bufferViewTask);
246  importTasks.Add(accessorTask);
247  GLTFImage.ImportTask imageTask = new GLTFImage.ImportTask(gltfObject.images, directoryRoot, bufferViewTask);
248  importTasks.Add(imageTask);
249  GLTFTexture.ImportTask textureTask = new GLTFTexture.ImportTask(gltfObject.textures, imageTask);
250  importTasks.Add(textureTask);
251  GLTFMaterial.ImportTask materialTask = new GLTFMaterial.ImportTask(gltfObject.materials, textureTask, importSettings);
252  importTasks.Add(materialTask);
253  GLTFMesh.ImportTask meshTask = new GLTFMesh.ImportTask(gltfObject.meshes, accessorTask, bufferViewTask, materialTask, importSettings);
254  importTasks.Add(meshTask);
255  GLTFSkin.ImportTask skinTask = new GLTFSkin.ImportTask(gltfObject.skins, accessorTask);
256  importTasks.Add(skinTask);
257  GLTFNode.ImportTask nodeTask = new GLTFNode.ImportTask(gltfObject.nodes, meshTask, skinTask, gltfObject.cameras);
258  importTasks.Add(nodeTask);
259 
260  // Ignite
261  for (int i = 0; i < importTasks.Count; i++) {
262  TaskSupervisor(importTasks[i], onProgress).RunCoroutine();
263  }
264 
265  // Wait for all tasks to finish
266  while (!importTasks.All(x => x.IsCompleted)) yield return null;
267 
268  // Fire onFinished when all tasks have completed
269  GameObject root = nodeTask.Result.GetRoot();
270  GLTFAnimation.ImportResult[] animationResult = gltfObject.animations.Import(accessorTask.Result, nodeTask.Result, importSettings);
271  AnimationClip[] animations = new AnimationClip[0];
272  if (animationResult != null) animations = animationResult.Select(x => x.clip).ToArray();
273  if (onFinished != null) onFinished(nodeTask.Result.GetRoot(), animations);
274 
275  // Close file streams
276  foreach (var item in bufferTask.Result) {
277  item.Dispose();
278  }
279  }

282  private static IEnumerator TaskSupervisor(ImportTask importTask, Action<float> onProgress = null) {
283  // Wait for required results to complete before starting
284  while (!importTask.IsReady) yield return null;
285  // Start threaded task
286  importTask.task.Start();
287  // Wait for task to complete
288  while (!importTask.task.IsCompleted) yield return null;
289  // Run additional unity code on main thread
290  importTask.OnCoroutine(onProgress).RunCoroutine();
291  //Wait for additional coroutines to complete
292  while (!importTask.IsCompleted) { yield return null; }
293  }

297  if (gLTFObject.extensionsRequired != null) {
298  for (int i = 0; i < gLTFObject.extensionsRequired.Count; i++) {
299  switch (gLTFObject.extensionsRequired[i]) {

301  break;

303  break;
304  default:

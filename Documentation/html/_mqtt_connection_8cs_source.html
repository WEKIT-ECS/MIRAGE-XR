77  public void Connect(string url, int port, string userName = "", string password = "", string clientId = "MQTTClient")
78  {
79  // Just to prevent identical MQTT client names...
80  var time = DateTime.UtcNow;
81  clientId = clientId + "-" + time.Hour + "-" + time.Minute + "-" + time.Second + "-" + time.Millisecond;
82 
83  // For Hololens.
84  #if WINDOWS_UWP
85  _client = new MqttClient(url, port, false, MqttSslProtocols.None);
86 
87  // For all the other platforms.
88  #else
89  _client = new MqttClient(url, port, false, null, null, MqttSslProtocols.None);
90  #endif
91 
92  // Register to callbacks
93  _client.MqttMsgPublishReceived += ReceiveAsync;
94 
95  // ConnectAsync...
96  try
97  {
98  // If user name is not set, connect without authentication
99  if (string.IsNullOrEmpty(userName))
100  _client.Connect(clientId);
101 
102  // If user name is set, connect with authentication
103  else
104  {
105  // Username must contain also a password
106  if (string.IsNullOrEmpty(password))
107  throw new ArgumentNullException(userName + " does not contain a password.");
108 
109  _client.Connect(clientId, userName, password);
110  }
111 
112  // Launch connection established succesfully event.
113  OnConnectionEstablished?.Invoke(true);
114  }
115  catch
116  {
117  // Launch connection establishing failed event.
118  OnConnectionEstablished?.Invoke(false);
119  throw;
120  }

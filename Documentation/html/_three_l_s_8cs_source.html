14 using System.Collections;
15 using System.Collections.Generic;
16 using UnityEngine;
17 
18 // ThreeLS Class
19 internal class ThreeLS
20 {
21  // Lipsync frequency bins
22  private float[] refFBins = { 0, 500, 700, 3000, 6000 };
23  public float[] freqBins;
24  // Lipsync blendshape weights
25  public float[] LSbsw = { 0, 0, 0 };
26  // Lipsync parameters
27  public float threshold;
28  public float smoothness;
29  public float vocalTractFactor;
30  // Energies
31  public float[] energies = { 0, 0, 0, 0 };
32 
33  // Audio file
34  public AudioSource audioSignal;
35  // FFT parameters
36  private FFTWindow blackmannWindow;
37  public int fftSize = 1024;
38  public int fs; // Sampling frequency
39  private float[] spectrum;
40  private float[] smoothSpectrumRaw;
41  public float[] smoothSpectrum;
42 
43  // Constructor
44  public ThreeLS(AudioSource audioIn, float thres, float smooth, float vocalTract)
45  {
46  // Audio source
47  audioSignal = audioIn;
48  // Parameters
49  threshold = thres;
50  smoothness = smooth;
51  vocalTractFactor = vocalTract;

53  init();

55  // Empty constructor
56  public ThreeLS(AudioSource audioIn)
57  {
58  // Audio source
59  audioSignal = audioIn;
60  // Parameters
61  threshold = 0.5f;
62  smoothness = 0.6f;
63  vocalTractFactor = 1.0f;
64  // Initialize
65  init();
66  }

68  private void init()
69  {
70  // Init frequency bins
71  updateFrequencyBins(vocalTractFactor);
72  // Initialize FFT
73  spectrum = new float[fftSize / 2];
74  smoothSpectrumRaw = new float[fftSize / 2];
75  smoothSpectrum = new float[fftSize / 2];
76  // FS
77  fs = AudioSettings.outputSampleRate;
78  }
79 
80  // Update frequency bins
81  public void updateFrequencyBins(float InVocalTractFactor)
82  {
83  vocalTractFactor = InVocalTractFactor;
84  freqBins = new float[refFBins.Length];
85  for (int i = 0; i < refFBins.Length; i++)
86  {
87  freqBins[i] = refFBins[i] * vocalTractFactor;
88  }
89  }
90 
91 
92  // Update lipsync
93  public void updateLS()
94  {
95  if (!audioSignal)
96  return;
97  if (!audioSignal.isPlaying)
98  return;
99  // FFT
100  smoothFFT();
101  // Energies
102  binAnalysis();
103  // Lip values
104  lipAnalysis();
105  }
106 
107 
108  // Smooth spectrum
109  private void smoothFFT()
110  {
111  audioSignal.GetSpectrumData(spectrum, 0, FFTWindow.Blackman);
112  for (var i = 0; i < spectrum.Length; i++)
113  {
114  smoothSpectrumRaw[i] = smoothSpectrumRaw[i] * (smoothness) + spectrum[i] * (1 - smoothness);
115  smoothSpectrum[i] = 20 * Mathf.Log10(smoothSpectrumRaw[i]);
116  }
117  }
118 
119 
120  // Analyze energies
121  private void binAnalysis()
122  {
123  int halfFFTSize = fftSize / 2; // Half the FFT size
124 
125  // Energies inside bins
126  for (int bindInd = 0; bindInd < freqBins.Length - 1; bindInd++)
127  {
128  // Start and end of bins
129  int indxIn = Mathf.RoundToInt(freqBins[bindInd] * halfFFTSize / (fs / 2));
130  int indxEnd = Mathf.RoundToInt(freqBins[bindInd + 1] * halfFFTSize / (fs / 2));
131 
132  // Sum of frequency values
133  energies[bindInd] = 0;
134  for (int i = indxIn; i < indxEnd; i++)
135  {
136  // Data goes from -25 to -160?
137  float vv = threshold + (smoothSpectrum[i] + 20) / 140;
138  // Zeroes negative values
139  vv = vv > 0 ? vv : 0;
140 
141  energies[bindInd] += vv;
142  }
143  // Normalize (divide by number of samples)
144  energies[bindInd] /= (indxEnd - indxIn);
145  }
146  }
147 
148 
149  // Calculate lipsync blend shape weights
150  private void lipAnalysis()
151  {
152  // Kiss blend shape
153  float vv = (0.5f - (energies[2])) * 2;
154  if (energies[1] < 0.2f)
155  vv = vv * (energies[1] * 5.0f);
156  vv = Mathf.Clamp01(vv); // Clamp
157  LSbsw[0] = vv;
158 
159  // Lips closed blend shape
160  vv = energies[3] * 3;
161  vv = Mathf.Clamp01(vv); // Clamp
162  LSbsw[1] = vv;
163 
164  // Mouth open blend shape
165  vv = energies[1] * 0.8f - energies[3] * 0.8f;
166  vv = Mathf.Clamp01(vv);
167  LSbsw[2] = vv;
168  }
169 }

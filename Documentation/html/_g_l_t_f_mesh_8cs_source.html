61  public MeshData(GLTFMesh gltfMesh, GLTFAccessor.ImportResult[] accessors, GLTFBufferView.ImportResult[] bufferViews) {
62  name = gltfMesh.name;
63  if (gltfMesh.primitives.Count == 0) {
64  Debug.LogWarning("0 primitives in mesh");
65  } else {
66  for (int i = 0; i < gltfMesh.primitives.Count; i++) {
67  GLTFPrimitive primitive = gltfMesh.primitives[i];
68  // Load draco mesh
69  if (primitive.extensions != null && primitive.extensions.KHR_draco_mesh_compression != null) {
70  GLTFPrimitive.DracoMeshCompression draco = primitive.extensions.KHR_draco_mesh_compression;
71  GLTFBufferView.ImportResult bufferView = bufferViews[draco.bufferView];
72  GLTFUtilityDracoLoader loader = new GLTFUtilityDracoLoader();
73  byte[] buffer = new byte[bufferView.byteLength];
74  bufferView.stream.Seek(bufferView.byteOffset, System.IO.SeekOrigin.Begin);
75 
76  bufferView.stream.Read(buffer, 0, bufferView.byteLength);
77 
78  GLTFUtilityDracoLoader.MeshAttributes attribs = new GLTFUtilityDracoLoader.MeshAttributes(
79  primitive.extensions.KHR_draco_mesh_compression.attributes.POSITION ?? -1,
80  primitive.extensions.KHR_draco_mesh_compression.attributes.NORMAL ?? -1,
81  primitive.extensions.KHR_draco_mesh_compression.attributes.TEXCOORD_0 ?? -1,
82  primitive.extensions.KHR_draco_mesh_compression.attributes.JOINTS_0 ?? -1,
83  primitive.extensions.KHR_draco_mesh_compression.attributes.WEIGHTS_0 ?? -1,
84  primitive.extensions.KHR_draco_mesh_compression.attributes.COLOR_0 ?? -1
85  );
86 
87  //Mesh mesh = loader.LoadMesh(buffer, attribs);
88 
89  GLTFUtilityDracoLoader.AsyncMesh asyncMesh = loader.LoadMesh(buffer, attribs);
90  if (asyncMesh == null) Debug.LogWarning("Draco mesh couldn't be loaded");
91 
92  submeshTrisMode.Add(primitive.mode);
93 
94  // Tris
95  int vertCount = verts.Count();
96  submeshTris.Add(asyncMesh.tris.Reverse().Select(x => x + vertCount).ToList());
97 
98  verts.AddRange(asyncMesh.verts.Select(x => new Vector3(-x.x, x.y, x.z)));
99  normals.AddRange(asyncMesh.norms.Select(v => { v.x = -v.x; return v; }));
100  //tangents.AddRange(asyncMesh.tangents.Select(v => { v.y = -v.y; v.z = -v.z; return v; }));
101 
102  // Weights
103  if (asyncMesh.boneWeights != null) {
104  if (weights == null) weights = new List<BoneWeight>();
105  weights.AddRange(asyncMesh.boneWeights);
106  }
107 
108  // BlendShapes not supported yet
109  /* for (int k = 0; k < mesh.blendShapeCount; k++) {
110  int frameCount = mesh.GetBlendShapeFrameCount(k);
111  BlendShape blendShape = new BlendShape();
112  blendShape.pos = new Vector3[frameCount];
113  blendShape.norm = new Vector3[frameCount];
114  blendShape.tan = new Vector3[frameCount];
115  for (int o = 0; o < frameCount; o++) {
116  mesh.GetBlendShapeFrameVertices(k, o, blendShape.pos, blendShape.norm, blendShape.tan);
117  }
118  blendShapes.Add(blendShape);
119  } */

121  // UVs
122  if (asyncMesh.uv != null) {
123  if (uv1 == null) uv1 = new List<Vector2>();
124  uv1.AddRange(asyncMesh.uv.Select(x => new Vector2(x.x, -x.y)));
125  }
126  }
127  // Load normal mesh
128  else {
129  int vertStartIndex = verts.Count;
130  submeshVertexStart.Add(vertStartIndex);
131 
132  // Verts - (X points left in GLTF)
133  if (primitive.attributes.POSITION.HasValue) {
134  IEnumerable<Vector3> newVerts = accessors[primitive.attributes.POSITION.Value].ReadVec3().Select(v => { v.x = -v.x; return v; });
135  verts.AddRange(newVerts);
136  }
137 
138  int vertCount = verts.Count;
139 
140  // Tris - (Invert all triangles. Instead of flipping each triangle, just flip the entire array. Much easier)
141  if (primitive.indices.HasValue) {
142  submeshTris.Add(new List<int>(accessors[primitive.indices.Value].ReadInt().Reverse().Select(x => x + vertStartIndex)));
143  submeshTrisMode.Add(primitive.mode);
144  }
145 
147  if (primitive.attributes.NORMAL.HasValue) {
148  normals.AddRange(accessors[primitive.attributes.NORMAL.Value].ReadVec3().Select(v => { v.x = -v.x; return v; }));
149  }
150 
151  // Tangents - (X points left in GLTF)
152  if (primitive.attributes.TANGENT.HasValue) {
153  tangents.AddRange(accessors[primitive.attributes.TANGENT.Value].ReadVec4().Select(v => { v.y = -v.y; v.z = -v.z; return v; }));
154  }
155 
156  // Vertex colors
157  if (primitive.attributes.COLOR_0.HasValue) {
158  colors.AddRange(accessors[primitive.attributes.COLOR_0.Value].ReadColor());
159  }
160 
161  // Weights
162  if (primitive.attributes.WEIGHTS_0.HasValue && primitive.attributes.JOINTS_0.HasValue) {
163  Vector4[] weights0 = accessors[primitive.attributes.WEIGHTS_0.Value].ReadVec4();
164  Vector4[] joints0 = accessors[primitive.attributes.JOINTS_0.Value].ReadVec4();
165  if (joints0.Length == weights0.Length) {
166  BoneWeight[] boneWeights = new BoneWeight[weights0.Length];
167  for (int k = 0; k < boneWeights.Length; k++) {
168  NormalizeWeights(ref weights0[k]);
169  boneWeights[k].weight0 = weights0[k].x;
170  boneWeights[k].weight1 = weights0[k].y;
171  boneWeights[k].weight2 = weights0[k].z;
172  boneWeights[k].weight3 = weights0[k].w;
173  boneWeights[k].boneIndex0 = Mathf.RoundToInt(joints0[k].x);
174  boneWeights[k].boneIndex1 = Mathf.RoundToInt(joints0[k].y);
175  boneWeights[k].boneIndex2 = Mathf.RoundToInt(joints0[k].z);
176  boneWeights[k].boneIndex3 = Mathf.RoundToInt(joints0[k].w);
177  }
178  if (weights == null) weights = new List<BoneWeight>(new BoneWeight[vertCount - boneWeights.Length]);
179  weights.AddRange(boneWeights);
180  } else Debug.LogWarning("WEIGHTS_0 and JOINTS_0 not same length. Skipped");
181  } else {
182  if (weights != null) weights.AddRange(new BoneWeight[vertCount - weights.Count]);
183  }
184 
185  // UVs
186  ReadUVs(ref uv1, accessors, primitive.attributes.TEXCOORD_0, vertCount);
187  ReadUVs(ref uv2, accessors, primitive.attributes.TEXCOORD_1, vertCount);
188  ReadUVs(ref uv3, accessors, primitive.attributes.TEXCOORD_2, vertCount);
189  ReadUVs(ref uv4, accessors, primitive.attributes.TEXCOORD_3, vertCount);
190  ReadUVs(ref uv5, accessors, primitive.attributes.TEXCOORD_4, vertCount);
191  ReadUVs(ref uv6, accessors, primitive.attributes.TEXCOORD_5, vertCount);
192  ReadUVs(ref uv7, accessors, primitive.attributes.TEXCOORD_6, vertCount);
193  ReadUVs(ref uv8, accessors, primitive.attributes.TEXCOORD_7, vertCount);
194  }
195  }
196 
197  bool hasTargetNames = gltfMesh.extras != null && gltfMesh.extras.targetNames != null;
198  if (hasTargetNames) {
199  if (gltfMesh.primitives.All(x => x.targets.Count != gltfMesh.extras.targetNames.Length)) {
200  Debug.LogWarning("Morph target names found in mesh " + name + " but array length does not match primitive morph target array length");
201  hasTargetNames = false;
202  }
203  }
204  // Read blend shapes after knowing final vertex count
205  int finalVertCount = verts.Count;
206 
207  for (int i = 0; i < gltfMesh.primitives.Count; i++) {
208  GLTFPrimitive primitive = gltfMesh.primitives[i];
209  if (primitive.targets != null) {
210  for (int k = 0; k < primitive.targets.Count; k++) {
211  BlendShape blendShape = new BlendShape();
212  blendShape.pos = GetMorphWeights(primitive.targets[k].POSITION, submeshVertexStart[i], finalVertCount, accessors);
213  blendShape.norm = GetMorphWeights(primitive.targets[k].NORMAL, submeshVertexStart[i], finalVertCount, accessors);
214  blendShape.tan = GetMorphWeights(primitive.targets[k].TANGENT, submeshVertexStart[i], finalVertCount, accessors);
215  if (hasTargetNames) blendShape.name = gltfMesh.extras.targetNames[k];
216  else blendShape.name = "morph-" + blendShapes.Count;
217  blendShapes.Add(blendShape);
218  }
219  }
220  }
221  }

224  private Vector3[] GetMorphWeights(int? accessor, int vertStartIndex, int vertCount, GLTFAccessor.ImportResult[] accessors) {
225  if (accessor.HasValue) {
226  if (accessors[accessor.Value] == null) {
227  Debug.LogWarning("Accessor is null");
228  return new Vector3[vertCount];
229  }
230  Vector3[] accessorData = accessors[accessor.Value].ReadVec3().Select(v => { v.x = -v.x; return v; }).ToArray();
231  if (accessorData.Length != vertCount) {
232  Vector3[] resized = new Vector3[vertCount];
233  Array.Copy(accessorData, 0, resized, vertStartIndex, accessorData.Length);
234  return resized;
235  } else return accessorData;
236  } else return new Vector3[vertCount];
237  }

239  public Mesh ToMesh() {
240  Mesh mesh = new Mesh();
241  if (verts.Count >= ushort.MaxValue) mesh.indexFormat = UnityEngine.Rendering.IndexFormat.UInt32;
242  mesh.vertices = verts.ToArray();
243  mesh.subMeshCount = submeshTris.Count;
244  var onlyTriangles = true;
245  for (int i = 0; i < submeshTris.Count; i++) {
246  switch (submeshTrisMode[i]) {
247  case RenderingMode.POINTS:
248  mesh.SetIndices(submeshTris[i].ToArray(), MeshTopology.Points, i);
249  onlyTriangles = false;
250  break;
251  case RenderingMode.LINES:
252  mesh.SetIndices(submeshTris[i].ToArray(), MeshTopology.Lines, i);
253  onlyTriangles = false;
254  break;
255  case RenderingMode.LINE_STRIP:
256  mesh.SetIndices(submeshTris[i].ToArray(), MeshTopology.LineStrip, i);
257  onlyTriangles = false;
258  break;
259  case RenderingMode.TRIANGLES:
260  mesh.SetTriangles(submeshTris[i].ToArray(), i);
261  break;
262  default:
263  Debug.LogWarning("GLTF rendering mode " + submeshTrisMode[i] + " not supported.");
264  return null;
265  }
266  }
267 
268  mesh.colors = colors.ToArray();
269  if (uv1 != null) mesh.uv = uv1.ToArray();
270  if (uv2 != null) mesh.uv2 = uv2.ToArray();
271  if (uv3 != null) mesh.uv3 = uv3.ToArray();
272  if (uv4 != null) mesh.uv4 = uv4.ToArray();
273  if (uv5 != null) mesh.uv5 = uv5.ToArray();
274  if (uv6 != null) mesh.uv6 = uv6.ToArray();
275  if (uv7 != null) mesh.uv7 = uv7.ToArray();
276  if (uv8 != null) mesh.uv8 = uv8.ToArray();
277  if (weights != null) mesh.boneWeights = weights.ToArray();
278 
279  mesh.RecalculateBounds();
280 
281  // Blend shapes
282  for (int i = 0; i < blendShapes.Count; i++) {
283  mesh.AddBlendShapeFrame(blendShapes[i].name, 1f, blendShapes[i].pos, blendShapes[i].norm, blendShapes[i].tan);
284  }
285 
286  if (normals.Count == 0 && onlyTriangles)
287  mesh.RecalculateNormals();
288  else
289  mesh.normals = normals.ToArray();
290 
291  if (tangents.Count == 0 && onlyTriangles)
292  mesh.RecalculateTangents();
293  else
294  mesh.tangents = tangents.ToArray();
295 
296  mesh.name = name;
297  return mesh;
298  }

300  public void NormalizeWeights(ref Vector4 weights) {
301  float total = weights.x + weights.y + weights.z + weights.w;
302  float mult = 1f / total;
303  weights.x *= mult;
304  weights.y *= mult;
305  weights.z *= mult;
306  weights.w *= mult;
307  }

309  private void ReadUVs(ref List<Vector2> uvs, GLTFAccessor.ImportResult[] accessors, int? texcoord, int vertCount) {
310  // If there are no valid texcoords
311  if (!texcoord.HasValue) {
312  // If there are already uvs, add some empty filler uvs so it still matches the vertex array
313  if (uvs != null) uvs.AddRange(new Vector2[vertCount - uvs.Count]);
314  return;
315  }
316  Vector2[] _uvs = accessors[texcoord.Value].ReadVec2();
317  FlipY(ref _uvs);
318  if (uvs == null) uvs = new List<Vector2>(_uvs);
319  else uvs.AddRange(_uvs);
320  }

347  public override IEnumerator OnCoroutine(Action<float> onProgress = null) {
348  // No mesh
349  if (meshData == null) {
350  if (onProgress != null) onProgress.Invoke(1f);
351  IsCompleted = true;
352  yield break;
353  }
354 
355  Result = new ImportResult[meshData.Length];
356  for (int i = 0; i < meshData.Length; i++) {
357  if (meshData[i] == null) {
358  Debug.LogWarning("Mesh " + i + " import error");
359  continue;
360  }
361 
362  Result[i] = new ImportResult();
363  Result[i].mesh = meshData[i].ToMesh();
364  Result[i].materials = new Material[meshes[i].primitives.Count];
365  for (int k = 0; k < meshes[i].primitives.Count; k++) {
366  int? matIndex = meshes[i].primitives[k].material;
367  if (matIndex.HasValue && materialTask.Result != null && materialTask.Result.Count() > matIndex.Value) {
368  GLTFMaterial.ImportResult matImport = materialTask.Result[matIndex.Value];
369  if (matImport != null) Result[i].materials[k] = matImport.material;
370  else {
371  Debug.LogWarning("Mesh[" + i + "].matIndex points to null material (index " + matIndex.Value + ")");
372  Result[i].materials[k] = GLTFMaterial.defaultMaterial;
373  }
374  } else {
375  Result[i].materials[k] = GLTFMaterial.defaultMaterial;
376  }
377  }
378  if (string.IsNullOrEmpty(Result[i].mesh.name)) Result[i].mesh.name = "mesh" + i;
379  if (onProgress != null) onProgress.Invoke((float) (i + 1) / (float) meshData.Length);
380  yield return null;
381  }
382  IsCompleted = true;
383  }
